window.onload = function () {
    _id('ajax').addEventListener('submit', sendAjax);
};

function sendAjax (e) {
    e.preventDefault();

    var call = new Ajax ({
        url: 'script.php',
        params: {
            name: _qs('[name="name"]').value
        }
    });

    call.success(function (data) {
        _log(data);
    });
    call.error(function (data) {
        _log(data);
    });

    return false;
}
